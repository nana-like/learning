RegExp 빌더

| string과 정규표현식

          var str = "a";

          //regular expression literal
          //우리는 a를 찾고 있는 것이다
          var pattern = /a/;

          //regular expression object constructor
          //위에 /a/랑 똑같음
          var pattern = new RegExp('a');

RegExp로 하는 일

  • 필요한 정보만 추출 = exec
  • 원하는 정보가 있는 지 확인 = test
  • 특정 정보를 치환 = replace
          var pattern = /a/;
          pattern.exec('abcde');
            -> ['a']

          var pattern = /a./; //.은 문자를 나타냄 (아무거나)
          pattern.exec('abcde');
            -> ['ab']

          var pattern = /a/;
          pattern.exec('bcdef');
            -> null

          pattern.match()
          //exec랑 비슷함..

          var pattern = /a/;
          pattern.text('abcde');
            -> true
          pattern.text('bcdef');
            -> false

          var pattern = /a/;
          var str = 'abcde';
          str.replace(pattern, 'A');
            -> 'Abcde'

| 옵션

          var xi = /a/;
          'Abcde'.match(xi);
            -> null
          
          var oi = /a/i;
          'Abcde'.match(oi);
            -> ['A'] //대소문자 가리지 않음

          var xg = /a/;
          'abceda'.match(xg);
            -> ['a'];

          var og = /a/g;
          'abceda'.match(og);
            -> ['a', 'a']; //중복값 각각 해당

          var ig = /a/ig;
          'AabcdAa'.match(ig);
            -> ['A', 'a', 'A', 'a']; //동시에 사용 가능

| 캡처

          (\w+)\s(\w+)

          \w = A~Z / a~z / 0~9
          + = 수량자 Quantifier (하나 이상이라는 뜻)
          \s = space

          var pattern = (\w+)\s(\w+);
          var str = 'coding everybody';
          var result = str.replace(pattern, '$2, $1'); //$1= 첫번째 그룹 //$2 = 두번째 그룹
            -> result = everybody, coding
